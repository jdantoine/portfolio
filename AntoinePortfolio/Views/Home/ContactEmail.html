                                    $(document).ready(function () {
                                        $(".input-f").focus(function () {
                                            $(this).css("outline-color", "#388E8E");
                                        });
                                        $("#submit-f").click(function () {
                                            var one = parseInt($("input[name=1st-f]").val());
                                            var oneF = parseFloat($("input[name=1st-f]").val());
                                            var factorial = 1;
                                            if ($.trim($("#numbers-f").text()) != "") {
                                                $("#numbers-f").empty();
                                            }
                                            if (isNaN(one)) {
                                                $("#error-f").empty();
                                                $("#error-f").append("
You must submit a number!
");
                                            }
                                            else if (one < 0) {
                                                $("#error-f").empty();
                                                $("#error-f").append("
Your number must be positive!
");
                                            }
                                            else if ((oneF % 1) != 0) {
                                                $("#error-f").empty();
                                                $("#error-f").append("
You must enter a whole number!
");
                                            }
                                            else if (one === 1) {
                                                $("#error-f").empty();
                                                $("#numbers-f").append("
1! = 1.
");
                                            }
                                            else if (one === 0) {
                                                $("#error-f").empty();
                                                $("#numbers-f").append("
0! = 1..... for some odd reason...
");
                                            }
                                            else {
                                                if (one >= 0) {
                                                    $("#error-f").empty();
                                                    for (i = one; i > 1; i--) {
                                                        factorial *= i;
                                                    }
                                                    $("#numbers-f").prepend("
" + one + "! = " + factorial + ".
");
                                                }
                                            }
                                        });
                                        $("#again-f").click(function () {
                                            $("#numbers-f").empty();
                                            $("#error-f").empty();
                                            $("#form-f")[0].reset();
                                        });
                                    });

                                    $(document).ready(function () {
                                        $(".input-fb").focus(function () {
                                            $(this).css("outline-color", "#388E8E");
                                        });
                                        $("#submit-fb").click(function () {
                                            var one = parseInt($("input[name=1st-fb]").val());
                                            var two = parseInt($("input[name=2nd-fb]").val());
                                            var oneF = parseFloat($("input[name=1st-fb]").val());
                                            var twoF = parseFloat($("input[name=2nd-fb]").val());
                                            if ($.trim($("#numbers-fb").text()) != "") {
                                                $("#numbers-fb").empty();
                                            }
                                            if ((isNaN(one)) || (isNaN(two))) {
                                                $("#error-label").empty();
                                                $("#error-label").append("
You must submit 2 numbers!
");
                                            }
                                            else if (((one < 1) || (one > 100)) || ((two < 1) || (two > 100))) {
                                                $("#error-label").empty();
                                                $("#error-label").append("
Your numbers must be between 1-100!
");
                                            }
                                            else if (((oneF % 1) != 0) || ((twoF % 1) != 0)) {
                                                $("#error-label").empty();
                                                $("#error-label").append("
You must enter whole numbers!
");
                                            }
                                            else {
                                                if (((one >= 1) && (one <= 100)) && ((two >= 1) && (two <= 100))) {
                                                    $("#error-label").empty();
                                                    for (i = 1; i < 101; i++) {
                                                        if ((i % one === 0) && (i % two === 0)) {
                                                            $("#numbers-fb").append("
FizzBuzz
");
                                                        }
                                                        else if (i % one === 0) {
                                                            $("#numbers-fb").append("
Fizz
");
                                                        }
                                                        else if (i % two === 0) {
                                                            $("#numbers-fb").append("
Buzz
");
                                                        }
                                                        else {
                                                            $("#numbers-fb").append("
" + i + "
");
                                                        }
                                                    }
                                                }
                                            }
                                        });
                                        $("#start-again-fb").click(function () {
                                            $("#numbers-fb").empty();
                                            $("#error-label").empty();
                                            $("#form-fb")[0].reset();
                                        });
                                    });

                                    $(document).ready(function () {
                                        $(".input-p").focus(function () {
                                            $(this).css("outline-color", "#388E8E");
                                        });
                                        $("#submit-p").click(function () {
                                            var one = $("input[name=1st-p]").val().toUpperCase();
                                            var oneR = one.split('').reverse().join('');
                                            var numberCheck = parseInt(one);
       
                                            if ($.trim($("#answer-p").text()) != "") {
                                                $("#answer-p").empty();
                                            }
                                            if ((one === "")) {
                                                $("#error-p").empty();
                                                $("#error-p").append("
You did not submit anything!
");
                                            }
                                            else if (isNaN(numberCheck) === false) {
                                                $("#error-p").empty();
                                                $("#error-p").append("
Numbers are invalid!
");
                                            }
                                            else {
                                                $("#error-p").empty();
                                                if (one === oneR) {
                                                    $("#answer-p").append("
'" + one + "' is a palindrome!
");
                                                }
                                                else {
                                                    $("#answer-p").append("
'" + one + "' is not a palindrome.
");
                                                }
                                            }        
                                        });
                                        $("#again-p").click(function () {
                                            $("#answer-p").empty();
                                            $("#error-p").empty();
                                            $("#form-p")[0].reset();
                                        });
                                    });
